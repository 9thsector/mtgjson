| New Set Instructions |
\----------------------/
# Add new set to C.js
# Run buildSet.js TWICE (first pass gets data, second pass gets foreign langauge names)
node build/buildSet.js <expansion code>
node build/buildSet.js <expansion code>

# Run buildSet.js with -forcePrintings and a list of all codes output by 'Other Printings' at the end of the previous buildSet for our new set
# This will update printings for all other sets that have cards from this new set

# Ensure images are in mtgimage.com

# Now verify
node verify/generate.js <expansion code>

# Now Open in browser:
file:///mnt/compendium/DevLab/mtgjson/verify/verify.html

# Now generate web
cd web
node generate.js

# Test in browser
http://dev.mtgjson.com

# Deploy
cd deploy
./deploy.sh
